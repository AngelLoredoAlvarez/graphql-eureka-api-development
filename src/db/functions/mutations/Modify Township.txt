CREATE OR REPLACE FUNCTION eureka_public.modify_township(
    type_township TEXT,
    township TEXT,
    postal_code TEXT,
    id_town UUID,
    id UUID
) RETURNS eureka_public.township AS $$
    UPDATE
        eureka_public.township
    SET
        type_township = $1,
        township = $2,
        postal_code = $3,
        id_town = $4
    WHERE
        id = $5
    RETURNING
        *;
$$ LANGUAGE SQL VOLATILE STRICT;
